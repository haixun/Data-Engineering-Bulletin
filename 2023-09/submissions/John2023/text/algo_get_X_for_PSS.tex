\begin{algorithm}
\caption{Compute $I_{sync}$ for \FullName}\label{algo:get_X_for_PSS}
\begin{algorithmic}[1]
\REQUIRE queue capacity $L_0$ of \SeqShortName, $T$ workers 
\ENSURE value of $I_{sync}$ for \FullName
\IF{$L_0 / T < 1$}
    \RETURN $1$
\ELSE
    \RETURN $L_0 / T$
\ENDIF
\end{algorithmic}
\end{algorithm}
