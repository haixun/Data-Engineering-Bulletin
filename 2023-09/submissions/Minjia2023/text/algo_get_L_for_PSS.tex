\begin{algorithm}
\caption{Compute $L$ for \FullName}\label{minjia_algo:get_L_for_PSS}
\begin{algorithmic}[1]
\REQUIRE value of $K$, queue capacity $L_0$ of \SeqShortName, $T$ workers 
\ENSURE value of $L$ for \FullName
\IF{$L_0 < K$}
    \RETURN $L_0$
\ELSIF{$(L_0 / T) < K$)}
    \RETURN K
\ELSE
    \RETURN $L_0 / T$
\ENDIF
\end{algorithmic}
\end{algorithm}
